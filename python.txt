# sintassi python

x = 9    #inizializzazione della variabile x con un numero intero
print(x)  #stampa il numero 9

y = 9.8  #tipo di dato float
print(y)

s = "stringa"  #tipo di dato stringa
print(s)     

print(s[0])  #stampa il primo carattere della stringa

b = True    #tipo di dato boolean
print(b)

a = 0
print(bool(a)) #la funzione bool verifica il valore di verità di una variabile

z = input("scrivi un numero intero: ") #legge un numero da tastiera 
print(z)   #stampa il numero 

print(type(z))   #funzione che stampa il tipo di dato della variabile

z = int(z)  #casting, conversione di tipo, da stringa a numero intero

#sintassi per le liste
lista = ["stringa1", "stringa2","stringa3","stringa4","stringa5","stringa6","stringa7","stringa8","stringa9"] 
print(type(lista)) 

#sintassi della selezione
if x < z:     
    print("x è minore di z")
else:
    print("x è maggiore di z")


if x < z:   
    print("x è minore di z")
elif x == z:
     print("x è uguale a z")  
else:
    print("x è maggiore z")


#sintassi del for
for stringa in lista:  
    print(stringa)   #stampa tutte le stringhe della lista

for carattere in s:  
    print(carattere)    #stampa i caratteri della stringa


#sintassi del while
i = 0;
while (i <= x):  
    print(i)
    i = i+1       #stampa i numeri da 0 ad x

#sintassi per il range
for x in range(x):
    print(x)    #stampa i numeri nel range da 0 ad x-1


#sintassi per le funzioni
def funzione(var):  #passaggio dei paramenti
    print(var)

funzione(x)   #esecuzione della funzione, stampa x


def funz_p(a, b):  
    prod = a * b
    return prod   #valore di ritorno

p = funz_p(x, z)
