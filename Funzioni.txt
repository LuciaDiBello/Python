def funzione():
   x = 9   #variabile locale
   print(x)  

funzione()  # esecuzione della funzione: stampa la variabile locale
print(x) #segnala un errore


def F():
  x = 9
  def funz():  #definizione di un'altra funzione 
     print(x)  #stesso ambito di visibilità di F
  funz()  #esecuzione di funz()

F() #esecuzione di F, stampa x

   print(y)  #stampa la variabile globale
